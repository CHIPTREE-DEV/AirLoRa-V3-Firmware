from(bucket: "Testing_Lora")
  |> range(start: v.timeRangeStart, stop: v.timeRangeStop)
  |> filter(fn: (r) => r["_measurement"] == "CO" or r["_measurement"] == "CO2" or r["_measurement"] == "PM1.0" or r["_measurement"] == "PM2.5" or r["_measurement"] == "Temperature")
  |> aggregateWindow(every: v.windowPeriod, fn: mean, createEmpty: false)
  |> yield(name: "mean")
